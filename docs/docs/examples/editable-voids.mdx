() => {
  const components = {
    ...CONFIG.components,
    [EDITABLE_VOID]: EditableVoidElement,
  }

  const plugins = createPlugins([
    ...PLUGINS.basicNodes,
    createEditableVoidPlugin(),
  ], {
    components,
  });

  return (
    <Plate
      id="editable-voids"
      plugins={plugins}
      editableProps={CONFIG.editableProps}
      initialValue={VALUES.editableVoids}
    />
  );
}
