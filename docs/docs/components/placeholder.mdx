### Installation

```bash npm2yarn
npm install @udecode/plate-placeholder
```

### Usage

```ts live
() => {
  const components = withStyledPlaceHolders(CONFIG.components);
  const plugins = createPlugins(PLUGINS.basicNodes, { components });

  return (
    <>
      <HeadingToolbar>
        <BasicElementToolbarButtons />
      </HeadingToolbar>
      <Plate
        id="placeholder"
        plugins={plugins}
        editableProps={CONFIG.editableProps}
        initialValue={VALUES.placeholder}
      />
    </>
  );
}
```
