The mention plugin is using the [combobox plugin](docs/plugins/combobox).

### Installation

```bash npm2yarn
npm install @udecode/plate-combobox
npm install @udecode/plate-mention
npm install @udecode/plate-mention-ui
```

### Usage

```ts live
() => {
  const styledComponents = createPlateComponents();

  const Editor = () => {
    const plugins = [
      ...pluginsBasic,
      createComboboxPlugin(),
      createMentionPlugin(),
    ];

    return (
      <Plate
        id="mention"
        plugins={plugins}
        components={styledComponents}
        options={options}
        editableProps={editableProps}
        initialValue={initialValueMentions}
      >
        <MentionCombobox items={MENTIONABLES} id="mention" trigger="@" searchPattern={`[\\S\\s]+`} />

        <MentionCombobox items={MENTIONABLES} id="mention2" trigger="#" />
      </Plate>
    );
  };

  return <Editor />
}
```
