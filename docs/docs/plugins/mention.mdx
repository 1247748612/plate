The mention plugin is using the [combobox plugin](docs/plugins/combobox).

### Installation

```bash npm2yarn
npm install @udecode/plate-combobox
npm install @udecode/plate-mention
npm install @udecode/plate-mention-ui
```

### Usage

```ts live
() => {
  const components = {
    ...CONFIG.components,
    mention2: MentionElement, 
  };

  const plugins = [
    ...PLUGINS.basicNodes,
    createComboboxPlugin(),
    ...createMentionPlugins({ trigger: '@' }),
    ...createMentionPlugins({ trigger: '#', key: 'mention2' }),
    ...createMentionPlugins({ trigger: '/', id: 'mention3' }),
  ];

  return (
    <Plate
      id="mention"
      plugins={plugins}
      components={components}
      options={CONFIG.options}
      editableProps={CONFIG.editableProps}
      initialValue={VALUES.mentions}
    >
      <MentionCombobox items={CONFIG.mentionItems} />
      <MentionCombobox items={CONFIG.mentionItems} pluginKey="mention2" />
      <MentionCombobox items={[CONFIG.mentionItems[0], CONFIG.mentionItems[1]]} id="mention3" />
    </Plate>
  );
}
```
