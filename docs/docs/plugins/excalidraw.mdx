### Installation

```bash npm2yarn
npm install @udecode/plate-excalidraw
```

### Usage

```ts live
() => {
  const plugins = [
    ...PLUGINS.basicNodes,
    createExcalidrawPlugin(),
    createSelectOnBackspacePlugin({ allow: [ELEMENT_EXCALIDRAW] }),
  ];

  const components = {
    ...CONFIG.components,
    [ELEMENT_EXCALIDRAW]: ExcalidrawElement
  };

  return (
    <Plate
      id="excalidraw"
      plugins={plugins}
      components={components}
      options={CONFIG.options}
      editableProps={CONFIG.editableProps}
      initialValue={VALUES.excalidraw}
    />
  );
}
```
