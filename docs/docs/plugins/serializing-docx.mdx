### Installation

```bash npm2yarn
npm install @udecode/plate-docx-serializer
```

### Usage

#### Deserializer

```ts live
() => {
  const plugins = [
    ...PLUGINS.basicNodes,
    createImagePlugin(),
    createLinkPlugin(),
    createIndentListPlugin(),
    createIndentPlugin(),
    createTablePlugin(),
    createTodoListPlugin(),
    createMediaEmbedPlugin(),
    createHighlightPlugin(),
  ];``
    
  plugins.push(createDeserializeDocxPlugin({ plugins }));

  return (
    <Plate
      id="deserialize-docx"
      plugins={plugins}
      components={CONFIG.components}
      options={CONFIG.docxOptions}
      editableProps={CONFIG.editableProps}
      initialValue={VALUES.pasteHtml}
    />
  );
}
```
